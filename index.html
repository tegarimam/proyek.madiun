function doPost(e) { var lock = LockService.getScriptLock(); lock.tryLock(10000); // Mencegah bentrokan data jika dikirim bersamaan try { var doc = SpreadsheetApp.getActiveSpreadsheet(); var sheet = doc.getSheets()[0]; // Mengambil sheet pertama var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; var nextRow = sheet.getLastRow() + 1; var newRow = []; // Sesuaikan urutan kolom: Waktu, Nama Pekerjaan, Detail, Persentase newRow.push(new Date()); newRow.push(e.parameter.namaPekerjaan); newRow.push(e.parameter.detailPekerjaan); newRow.push(e.parameter.persentase + "%"); sheet.getRange(nextRow, 1, 1, newRow.length).setValues([newRow]); return ContentService .createTextOutput(JSON.stringify({ 'result': 'success', 'row': nextRow })) .setMimeType(ContentService.MimeType.JSON); } catch (f) { return ContentService .createTextOutput(JSON.stringify({ 'result': 'error', 'error': f })) .setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } }
